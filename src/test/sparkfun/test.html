  {680, 900},//ring\n
  {736, 907} //pinky\n
};\n
\n
void setup() {\n
\n
  Wire.begin();\n
  Serial.begin(115200);\n
\n
  //Begin our finger sensors, change addresses as needed.\n
  if (pinkySensor.begin(Wire, 100000, ADS1015_ADDRESS_SDA) == false) \n
  {\n
     Serial.println("Pinky not found. Check wiring.");\n
     while (1);\n
  }\n
  if (indexSensor.begin(Wire, 100000, ADS1015_ADDRESS_GND) == false) \n
  {\n
     Serial.println("Index not found. Check wiring.");\n
     while (1);\n
  }\n
\n
  //Set the calibration values for the hand.\n
  for (int channel; channel < 2; channel++)\n
  {\n
    for (int hiLo = 0; hiLo < 2; hiLo++)\n
    {\n
      indexSensor.setCalibration(channel, hiLo, handCalibration[channel][hiLo]);\n
      pinkySensor.setCalibration(channel, hiLo, handCalibration[channel + 2][hiLo]);\n
    }\n
    Serial.println();\n
  }\n
}\n
\n
void loop() {\n
  for (int channel = 0; channel < 2; channel++)\n
  {\n
    //Keep in mind that getScaledAnalogData returns a float\n
    hand[channel] = indexSensor.getScaledAnalogData(channel);\n
    hand[channel + 2] = pinkySensor.getScaledAnalogData(channel);\n
  }\n
  for (int finger = 0; finger < 4; finger++)\n
  {\n
    Serial.print(finger);\n
    Serial.print(": ");\n
    Serial.print(hand[finger]);\n
    Serial.print(" ");\n
  }\n
  Serial.println();\n
}\n
